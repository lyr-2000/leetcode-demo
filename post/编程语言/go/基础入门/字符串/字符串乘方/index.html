package main
import (
    "fmt"
    // "bufio"
    // "os"
    "strings"
)

func main() {
    // std := bufio.NewReader(os.Stdin)
    for {
        var s string 
        fmt.Scanf("%s",&s)
        if s == "." { break}
        //反求 a,使得 {a,a,a...}{n个} => s
        //n最大，即 a最短
        n := len(s)
        ret := 1
        for i:=1;i<=n;i++ {
            if n %i != 0 { continue}
            t := string([]rune(s)[0:i])
            cnt := n/i 
            k := strings.Repeat(t,cnt)
            if k == s {
                ret = cnt 
                break
            }
            //fmt.Println(k+s)
            
        }
    
        fmt.Println(ret)
         
        
        
    }
    
}
