        fmt.Print("=")

package main
import (
    "os"
    "bufio"
    "strings"
    "fmt"
)
func main() {
    r := bufio.NewReader(os.Stdin)
    f,_ := r.ReadString('\n')
    s := false
    var b strings.Builder 
    for _,v := range f {
        if v == '\n' { continue}
        if v == ' ' {
            if s {
                continue
            }
            s = true
            b.WriteRune(v)
        }else {
            s = false 
            b.WriteRune(v)
        }
    }
    fmt.Println(b.String())
}

package main
import (
    "fmt"
    "bufio"
    "os"
    "strings"
)

func main() {
    r := bufio.NewReader(os.Stdin)
    f,_ := r.ReadString('\n')
    x,_ := r.ReadString('\n')
    y,_ := r.ReadString('\n')
    
    f,x,y = strings.TrimRight(f,"\n"), strings.TrimRight(x,"\n"),strings.TrimRight(y,"\n")
    arr := strings.Fields(f)
    for _,v := range arr {
        if v==x {
            v = y
        }
        fmt.Printf("%s ",v)
    }
    
}

package main 

import (
    "fmt"
)


func main() {
    var s string 
    var t = 0
    fmt.Scanf("%d",&t)
    for t > 0 {
        t-- 
        fmt.Scanf("%s",&s)
        var pre = '\n'
        var cnt = 0
        var maxcnt = 0
        var maxres rune
        for _,v := range s {
            if v == pre {
                cnt++
            }else {
                pre = v
                cnt = 1
            }
            if maxcnt < cnt {
                maxres = pre
                maxcnt = cnt
            }
        }
        fmt.Printf("%c %d\n",maxres,maxcnt)
    }
    
}
