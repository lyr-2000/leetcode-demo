func dfs(g [][]int, node, color int, p []int) bool {
	// n := len(g)
	p[node] = color
	// target := color
	target := 2
	if color == 2 {
		target = 1
	}
	arr := g[node]
	for _, v := range arr {
		if p[v] == 0 {
			if !dfs(g, v, target, p) {
				return false
			}
		} else if color == p[v] { //conflicts color
			return false
		}
	}
	return true

}

func isBipartite(graph [][]int) bool {
	//color[i]==0, set color
	//else if color[i]== preColor { return false }
	// return true
	n := len(graph)
	paint := make([]int, n)
	for i := range graph {
		if paint[i] == 0 {
			if !dfs(graph, i, 1, paint) {
				return false
			}
		}
	}

	return true
}

package main

import "fmt"

func isPrime(x int) bool {
	if x < 2 {
		return false
	}
	for i := 2; i*i <= x; i++ {
		if x%i == 0 {
			return false
		}
	}
	return true
}
func matchit(curr int, all []int, match map[int]int, vis []bool) bool {
	for i, v := range all {
		if !vis[i] {
			if isPrime(curr + v) {
				vis[i] = true
				if match[i] == 0 || matchit(match[i], all, match, vis) {
					match[i] = curr
					return true
				}
			}
		}
	}
	return false
}
func main() {
	var (
		n int
	)
	fmt.Scan(&n)
	var (
		t1 []int
		t2 []int
	)
	for i := 0; i < n; i++ {
		var t int
		fmt.Scan(&t)
		if t%2 == 0 {
			t1 = append(t1, t)
		} else {
			t2 = append(t2, t)
		}
	}
	var total int
	match := make(map[int]int)

	vis := make([]bool, len(t2))
	for _, v := range t1 {
		if matchit(v, t2, match, vis) {
			total++
		}
		for i := range vis {
			vis[i] = false
		}
	}
	fmt.Println(total)
}
