考试题目和要点：

1、中文大写金额数字前应标明“人民币”字样。中文大写金额数字应用壹、贰、叁、肆、伍、陆、柒、捌、玖、拾、佰、仟、万、亿、元、角、分、零、整等字样填写。

2、中文大写金额数字到“元”为止的，在“元”之后，应写“整字，如532.00应写成“人民币伍佰叁拾贰元整”。在”角“和”分“后面不写”整字。

3、阿拉伯数字中间有“0”时，中文大写要写“零”字，阿拉伯数字中间连续有几个“0”时，中文大写金额中间只写一个“零”字，如6007.14，应写成“人民币陆仟零柒元壹角肆分“。
4、10应写作“拾”，100应写作“壹佰”。例如，1010.00应写作“人民币壹仟零拾元整”，110.00应写作“人民币壹佰拾元整”
5、十万以上的数字接千不用加“零”，例如，30105000.00应写作“人民币叁仟零拾万伍仟元整”

package main

import (
	"fmt"
	"strings"
)

var ss = strings.Split("零壹贰叁肆伍陆柒捌玖", "")

func maprow(d int) string {
	return ss[d]
}
func mapunit(sign, idx int, tot int) string {
	if sign < 0 {
		return string([]rune("角分")[idx])
	}
	num := [][]string{
		{""},
		{"拾"},
		{"佰"},
		{"仟"},
		{"万"},
		{"拾"},
		{"佰"},
		{"仟"},
		{"亿"},
		{"拾"},
		{"佰"},
		{"仟"},
		{"万"},
	}

	return num[tot-idx-1][0]

}

func decode(left, right string) string {
	var (
		suffix string
		prefix string
	)
	for i, v := range right {
		if v == '0' {
			continue
		}
		suffix += maprow(int(v-'0')) + mapunit(-1, i, 0)
	}
	if left != "0" {
		for i, v := range left {
			prefix += maprow(int(v-'0')) + mapunit(1, i, len(left))
		}
	}

	if len(left) >= 2 && prefix != "" {

		prefix = strings.TrimPrefix(prefix, "壹") + "元"
	} else if prefix != "" {
		prefix += "元"
	}
	// 壹拾万 伍万 壹仟 壹佰 贰拾 壹 壹角伍分
	//15万 1
	return "人民币" + prefix + suffix
}

func main() {
	var s string
	fmt.Scan(&s)
	all := strings.Split(s, ".")
	if len(all) == 1 {
		all = append(all, "")
	}
	fmt.Println(decode(all[0], all[1]))
}
