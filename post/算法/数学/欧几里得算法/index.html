欧几里得算法

辗转相除法的意义：
把 一个大规模的问题转化为一个小规模的问题。

辗转相除法用来求解 整数 a,b 的最大公约数, 表示 gcd(a,b), gcd(a,b) = gcd(b,a%b)

证明1：

b 和 a%b 的最大公约数，是a和b的公约数

b 和 a%b 的最大公约数，也是 a和b的最大公约数

#include <bits/stdc++.h>

using namespace std;
int gcd(int a,int b) {
    if(b) return gcd(b,a%b);
    return a;
}


void printGcd(int a,int b) {
    printf("gcd(%d,%d) = %d\n",a,b,gcd(a,b));
}

int main()
{
    vector<vector<int> >  a = {
        {2,4},
        {4,6},
        {4,2},
        {2,0},
    };
    for (auto &v : a) {
        printGcd(v[0],v[1]);
    }

    int n;
}
/*
gcd(2,4) = 2
gcd(4,6) = 2
gcd(4,2) = 2
gcd(2,0) = 2
*/

证明

1. b 和 a%b 的最大公约数，是a和b的公约数

设 b, a%b 的最大公约数是 c, 所以

$b = k_1 *c$

$a%b = k_2* c$

$a=k_2 c + kb = k_2c + kk_1*c$

所以 b 和 a%b 的最大公约数 也是 a和 b的最大公约数

b 和 a%b 的最大公约数，是a和b的最大公约数

$a=(k2 + k_1*k)*c$

$b = k_1 * c$

反证法：
假设c不是 a和 b的最大公约数， $gcd(k2+k_1*k,k_1) = d
\ d!=1$

贝祖等式

$ax+by = gcd(a,b) = c$

贝祖等式能保证式子一定有解

int gcd(int a, int b)
{
    if (b)
        return gcd(b, a % b);
    return a;
}

int is_prime(int x)
{
    if(x<=1) return false;
    for (int i = 2; i * i <= x; i++)
    {
        if (x % i == 0)
            return 0;
    }
    return 1;
}

int main()
{
    int n, m, ans = 0;
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= m; j++)
        {
            if (!is_prime(gcd(i, j)))
                continue;
            ans++;
            cout << i << " " << j << endl;
        }
    }
    cout << "total=" << ans << endl;
}
