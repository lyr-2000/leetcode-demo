class Solution {
public:
    string longestPalindrome(string s) {
        int n = s.size();
        vector<vector<int>> dp(n+1,vector<int>(n+1));
        dp[0][0] = 1;
        for(int i=0;i<n;i++) dp[i][i] =1;
        for(int i=n-1;i>=0;i--) {
            for(int j=i+1;j<n;j++) {
                if(s[i] != s[j]) {
                    dp[i][j] = false;
                    continue;
                }
                // j>i 
                if(j-i == 2) {
                    //aba 
                    dp[i][j] = s[i] == s[j];
                    continue;
                }
                if(j-i==1) {
                    //ab
                    dp[i][j] = s[i]==s[j];
                    continue;
                }
                dp[i][j] = dp[i+1][j-1];
            }
        }
        int l=0,r=0,mx=-1;
        // 最长 
        for(int i=0;i<n;i++) {
            for(int j=i;j<n;j++) {
                if(dp[i][j] && j-i + 1 > mx) {
                    l=i,r=j;
                    mx = r-l+1;
                }
            }
        }
        return s.substr(l,r-l+1);
    }
};

‘.’ 匹配任意单个字符
‘*’ 匹配零个或多个前面的那一个元素
所谓匹配，是要涵盖 整个 字符串 s 的，而不是部分字符串。

class Solution {
public:
    bool isMatch(string s, string p) {
        int n = s.size();
        int m = p.size();  // 这里要修正为 p 的大小
        vector<vector<bool>> dp(n + 1, vector<bool>(m + 1, false));  // 使用 bool 类型
        dp[0][0] = true;  // 空字符串与空模式匹配
        
        // 处理 p 中的模式可以匹配空字符串的情况
        for (int j = 1; j <= m; j++) {
            if (p[j - 1] == '*') {
                if(j>=2)
                    dp[0][j] = dp[0][j - 2];  // 例如 a*、.* 等
            }
        }

        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                if (p[j-1] == s[i-1] || p[j-1] == '.') {
                    dp[i][j] = dp[i-1][j-1];
                }else if(p[j-1] == '*') {
                    //match previous char 
                    if(p[j-2] == '.' || s[i-1] == p[j-2]) {
                        //匹配   0个， 匹配1~N
                        dp[i][j] = dp[i][j-2] || dp[i-1][j] ;
                    }else {
                        // p 前一个不能 和 当前的s 比较, 匹配0 个字符
                        dp[i][j] = dp[i][j-2];
                    }
                } 
            }
        }
        return dp[n][m];
    }
