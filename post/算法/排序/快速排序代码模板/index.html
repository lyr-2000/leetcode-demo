void quick_sort(int *arr,int l,int r) {
    if (l>=r) return ;
    int x = arr[l];
    int bl = l,br = r;
    while(l<r) {
        while(l < r && arr[r] >= x) r--; //>=
        if (l < r ) arr[l++] = arr[r];
        while(l < r && arr[l] < x) l++; //<
        if(l<r) arr[r--] = arr[l];
    }
    arr[l] = x;
    quick_sort(arr,bl,l-1);
    quick_sort(arr,l+1,br);
}   

int main(void)
{
    int arr[] = {1,5 ,2,2,111,6666,-11111,77,22,4355,4577,0,856678568,-1,412};
    int N = sizeof(arr)/sizeof(int);
    quick_sort(arr,0,N-1);
    for (int i=0;i<N;i++) {
        printf("%d,",arr[i]);
    }
    return 0;
}

#include <iostream>

using namespace std;

const int N = 1e5+10;
int a[N];


int fast_select (int l,int r, int k) {
    if (l>=r) return a[l];
    int mid = l+r >> 1;
    int i = l-1, j = r+1, x = a[mid];
    while (i<j) {
        do ++i; while (i<=r  && a[i] < x);
        do --j; while (j>=l && a[j]>x);
        if (i<j) swap (a[i], a[j]);
    }
    int index = j-l+1;
    if (index >=k) return fast_select(l,j,k);
    return fast_select(j+1, r,k-index);
}

int main() {
    int n,k;cin>>n>>k;
    for (int i=0;i<n;++i) cin>>a[i];
    
    cout << fast_select(0,n-1,k);
    
    
}

package main 
/**
 * 找到数组中第 K 大的元素
 * @param a int整型一维数组
 * @param n int整型 数组长度
 * @param K int整型 第 K 大
 * @return int整型
 */
func findKth(a []int, n int, K int) int {
    // 第 K 大元素等价于数组中第 n-K+1 小的元素
    targetIndex := n - K
    return quickSelect(a, 0, n-1, targetIndex)
}

// 快速选择函数
func quickSelect(a []int, left, right, k int) int {
    if left == right {
        return a[left]
    }

    pivotIndex := partition(a, left, right)

    if pivotIndex == k {
        return a[k]
    } else if pivotIndex > k {
        return quickSelect(a, left, pivotIndex-1, k)
    } else {
        return quickSelect(a, pivotIndex+1, right, k)
    }
}

// 分区函数，返回分区后基准值的最终位置
func partition(a []int, left, right int) int {
    pivot := a[right] // 选择最右边的元素作为基准值
    i := left - 1

    for j := left; j < right; j++ {
        if a[j] <= pivot {
            i++
            a[i], a[j] = a[j], a[i]
        }
    }
    // 将基准值放到正确的位置
    a[i+1], a[right] = a[right], a[i+1]
    return i + 1
}
