1268. 搜索推荐系统

Description

Difficulty: 中等

Related Topics: 字典树 , 数组 , 字符串

给你一个产品数组 products 和一个字符串 searchWord ，products  数组中每个产品都是一个字符串。

请你设计一个推荐系统，在依次输入单词 searchWord 的每一个字母后，推荐 products 数组中前缀与 searchWord 相同的最多三个产品。如果前缀相同的可推荐产品超过三个，请按字典序返回最小的三个。

请你以二维列表的形式，返回在输入 searchWord 每个字母后相应的推荐产品的列表。

struct Node{
    Node *next[26];
    bool flag;
    Node() {
        flag = false;
        for(int i=0;i<26;i++) next[i] = NULL;
    }
    ~Node() {}

    set<string> s;

};
class Trie {
public:
    Trie() {
        root = new Node();
    }
    Node *root;

    
    void insert(string word) {
        Node *p = root;
        for (auto x: word) {
            int ind = x-'a';
            if(p->next[ind] == NULL) 
                p->next[ind] = new Node();
            p = p->next[ind];
            p -> s.insert(word);
            if(p -> s.size() > 3) {
                auto iter = p->s.end();
                iter--;
                p -> s.erase(iter);
            }
        }
        p->flag= true;
        return;
    }
    
    vector<vector<string>> search(string word) {
        vector<vector<string>> ret;
        Node *p = root;
        for (auto x: word) {
            int ind = x-'a';
            if(p == NULL)  {
                ret.push_back(vector<string>());
                continue;
            }

            p = p->next[ind];
            vector<string> tmp;
            if (p) {
                for(auto s: p->s) {
                    tmp.push_back(s);
                }
            }
            ret.push_back(tmp);
        }
        return ret;
    }
    bool startsWith(string prefix) {
Node *p = root;
        for (auto x: prefix) {
            int ind = x-'a';
            if(p->next[ind] == NULL)  return false;
                // p->next[ind] = new Node();
            p = p->next[ind];
        }
        return p!=NULL;
    }
};
class Solution {
public:
    vector<vector<string>> suggestedProducts(vector<string>& products, string searchWord) {
        Trie tree;
        for(auto &s: products) {
            tree.insert(s);
        }
        return tree.search(searchWord);
    }
};
