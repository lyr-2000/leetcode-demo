23. 合并K个升序链表

Description

Difficulty: 困难

Related Topics: 链表 , 分治 , 堆（优先队列） , 归并排序

给你一个链表数组，每个链表都已经按升序排列。

请你将所有链表合并到一个升序链表中，返回合并后的链表。

// @lc code=start
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution
{
public:
     

     
    ListNode *mergeKLists(vector<ListNode *> &lists)
    {
       if(lists.size() == 0) return NULL;
       #define HeapItem pair<int,ListNode*>
       struct Compare {
           bool operator() (HeapItem &l,HeapItem &r) {
               return l.first >= r.first;
           };
       };
       priority_queue<HeapItem,vector<HeapItem>,Compare> q;
       for(auto lstNode: lists) {
           if(lstNode) q.push({lstNode->val,lstNode});
       }
       ListNode head;
       ListNode* p = &head;
       while(q.size()) {
           auto temp = q.top();q.pop();
           ListNode* lst = temp.second;
           p->next = lst;
           p = p->next;
           if(lst->next) {
               q.push({lst->next->val,lst->next});
           }
           
       }
       
       return head.next;


        
    }
};

    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if(lists.size()<=0) 
            return NULL;
        priority_queue<Node > q;
        for (ListNode *temp : lists) {
            if (temp) q.push ({temp->val,temp});
        }
        ListNode fake;
        ListNode *p = &fake;
        while(!q.empty()) {
            auto pnode = q.top();q.pop();
            p->next = pnode.node;
            // p = p->next;
            p = p->next;
            if(pnode.node->next) {
                q.push({pnode.node->next->val,pnode.node->next});
            }
        }
        return fake.next;
    }
};

#define Node ListNode
class Heap {
public:
    vector<Node*> heap;
    void push(Node* node) {
        heap.push_back(node);
        int p = heap.size() -1 ;
        while (p > 0) {
            int fa = (p-1)/2;
            if (heap[p]->val < heap[fa] ->val ) {
                swap(heap[p],heap[fa]);
                p = fa;
            }else {
                break;
            }
        }

    }
    Node* pop() {
        Node * res = heap[0];
        swap(heap[heap.size()-1],heap[0]);
        heap.pop_back();
        int  p = 0;
        //向下调整
        int child = p*2 + 1;
        // int rchild = p*2+2;
        while(child < heap.size()) {
            //child 未出界
            int rchild = p*2+2;
            if(rchild< heap.size() && heap[child]->val > heap[rchild]->val) {
                child = rchild;
            }
            if (heap[p]->val > heap[child]->val) {
                 swap(heap[child],heap[p]);
                p = child;
                child = p*2+1;
            }else {
                break;
            }
        }
        return res;
    }
};
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if(lists.size()<=0) 
            return NULL;
        Heap heap;
        for (Node * n:lists) {
            if(n) heap.push(n);
        }
        Node fake = {-1,NULL};

        Node *curr = &fake;
        while(heap.heap.size()) {
            
            Node * x = heap.pop();
            // if (!x) break;
            if (!x) continue;
            curr->next = x;
            curr = curr->next;
            if(x->next) {
                heap.push(x->next);
            }
        }
        return fake.next;
    }
};
