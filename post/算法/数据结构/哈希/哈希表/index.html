class HashTable
{
public:
    vector<bool> flag;
    vector<string> data;
    int cnt;
    HashTable(int n = 100) : flag(n), data(n), cnt(n) {}
    bool find(string &s)
    {
        int ind = hash_func(s) % data.size();
        recalc_ind(ind, s);
        return flag[ind];
    }
    bool remove(string &s)
    {
        int ind = hash_func(s) % data.size();
        recalc_ind(ind, s);
        flag[ind] = false;
        --cnt;
    }
    void insert(string &s)
    {
        int ind = hash_func(s) % data.size();
        recalc_ind(ind, s);
        if (!flag[ind])
        {
            data[ind] = s;
            flag[ind] = true;
            cnt += 1;
            if (cnt * 100 > data.size() * 75)
            {
                expand();
            }
        }
    }
    void expand()
    {

        vector<int> newFlag;
        int n = data.size();
        HashTable newTab(n * 2);
        for (int i = 0; i < n; i++)
        {
            if (!flag[i])
                continue;
            newTab.insert(data[i]);
        }
        *this = newTab;
    }
    int hash_func(string &s)
    {
        int seed = 131, hash = 0;
        for (int i = 0; s[i]; i++)
        {
            hash = hash * seed + s[i];
        }
        return hash & 0x7fffffff;
    }
    void recalc_ind(int ind, string &s)
    {
        int t = 1;
        while (flag[ind] && data[ind] != s)
        {
            ind += t * t;
            t++;
            ind %= data.size();
        }
    }
};
