/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func removeNthFromEnd(head *ListNode, n int) *ListNode {
    var h ListNode
    h.Next = head 
    l,r := &h,head 
    for  range n {
        r= r.Next 
    }
    for r != nil {
        l = l.Next 
        r = r.Next 
    }
    l.Next = l.Next.Next 
    return h.Next 
}
