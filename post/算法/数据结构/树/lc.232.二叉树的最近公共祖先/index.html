236. 二叉树的最近公共祖先

Description

Difficulty: 中等

Related Topics: 树, 深度优先搜索, 二叉树

给定一个二叉树, 找到该树中两个指定节点的最近公共祖先。

百度百科中最近公共祖先的定义为：“对于有根树 T 的两个节点 p、q，最近公共祖先表示为一个节点 x，满足 x 是 p、q 的祖先且 x 的深度尽可能大（一个节点也可以是它自己的祖先）。”

示例 1：

1
2
3

输入：root = [3,5,1,6,2,0,8,null,null,7,4], p = 5, q = 1
输出：3
解释：节点 5 和节点 1 的最近公共祖先是节点 3 。

示例 2：

1
2
3

输入：root = [3,5,1,6,2,0,8,null,null,7,4], p = 5, q = 4
输出：5
解释：节点 5 和节点 4 的最近公共祖先是节点 5 。因为根据定义最近公共祖先节点可以为节点本身。

示例 3：

1
2

输入：root = [1,2], p = 1, q = 2
输出：1

提示：

  • 树中节点数目在范围 [2, 105] 内。
  • -109 <= Node.val <= 109
  • 所有 Node.val 互不相同 。
  • p != q
  • p 和 q 均存在于给定的二叉树中。

思路解析

使用向上标记法， 有 a,b 两个节点，a从下面往上面染色， 染完根节点之后，再从b出发往上走，走到第一个染色节点，这个节点就是 a,b两个节点的 LCA

因为标记的时候每个节点只会走一次，时间复杂度是 $O(N)$

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
 #define Node TreeNode
class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        unordered_map<int,Node*>mp;
        mark(mp,root,NULL);
        remove_mark(mp,p);
        return get_no_mark(mp,q);
    }
    Node* get_no_mark(unordered_map<int,Node*> &mp,TreeNode *q) {
        if(q==NULL || !mp.count(q->val)) {
            return q;
        }
        Node *parent = mp[q->val];
        if(parent == NULL) 
            return q;
        return get_no_mark(mp,parent);
    }
    void remove_mark(unordered_map<int,Node*> &mp, TreeNode *p) {
        if(p==NULL) return;
        Node* parent = mp[p->val];
        mp.erase(p->val);
        if(parent) remove_mark(mp,parent);
    }
    void mark(unordered_map<int,Node*> &mp, TreeNode*p,TreeNode*parent) {
        if(p == NULL) return ;

        mp[p->val] = parent;
        if(p->left) mark(mp,p->left,p);
        if(p->right) mark(mp,p->right,p);
    }
};

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
 #define Node TreeNode
class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if(p==NULL || q == NULL||root==NULL) return root;
        unordered_map<int,Node*>mp;
        Node dummy(0);
        mark(mp,root,&dummy);
        // remove_mark(mp,p);
        // return get_no_mark(mp,q);
        unordered_map<int,Node*> redNodes;
        redNodes[root->val] = root;

        while( p->val!=root->val) {
            redNodes[p->val] = p;
            p = mp[p->val];
            if(p==NULL) break;
        }
        while(!redNodes.count(q->val)) {
            q = mp[q->val];
        }
        return q;


    }
    Node* get_no_mark(unordered_map<int,Node*> &mp,TreeNode *q) {
        if(q==NULL || !mp.count(q->val)) {
            return q;
        }
        Node *parent = mp[q->val];
        if(parent == NULL) 
            return q;
        return get_no_mark(mp,parent);
    }
    void remove_mark(unordered_map<int,Node*> &mp, TreeNode *p) {
        if(p==NULL) return;
        Node* parent = mp[p->val];
        mp.erase(p->val);
        if(parent) remove_mark(mp,parent);
    }
    void mark(unordered_map<int,Node*> &mp, TreeNode*p,TreeNode*parent) {
        if(p == NULL) return ;

        mp[p->val] = parent;
        if(p->left) mark(mp,p->left,p);
        if(p->right) mark(mp,p->right,p);
    }
};
