473. 火柴拼正方形

Description

Difficulty: 中等

Related Topics: 位运算 , 数组 , 动态规划 , 回溯 , 状态压缩

你将得到一个整数数组 matchsticks ，其中 matchsticks[i] 是第 i 个火柴棒的长度。你要用 所有的火柴棍 拼成一个正方形。你 不能折断 任何一根火柴棒，但你可以把它们连在一起，而且每根火柴棒必须 使用一次 。

如果你能使这个正方形，则返回 true ，否则返回 false 。

示例 1:

1
2
3

输入: matchsticks = [1,1,2,2,2]
输出: true
解释: 能拼成一个边长为2的正方形，每边两根火柴。

示例 2:

1
2
3

输入: matchsticks = [3,3,3,3,4]
输出: false
解释: 不能用所有火柴拼成一个正方形。

提示:

  • 1 <= matchsticks.length <= 15
  • 1 <= matchsticks[i] <= 108

Language: C++

class Solution {
public:
    bool makesquare(vector<int>& matchsticks) {
        int n = matchsticks.size();
        if (n<4) return false;

        sort(matchsticks.begin(),matchsticks.end());
        vector<int> arr(4);
        int sum = 0;
        for (auto x: matchsticks) sum += x; //计算木棍总长度

        for (int i=0;i<4;i++) arr[i] = sum / 4;
        if (arr[0] * 4 != sum) return false;
        return dfs(matchsticks.size()-1, arr,matchsticks);
    }

    bool dfs(int ind,vector<int> &arr,vector<int> &matchs) {
        if (ind == -1) return true;
        //尝试第 i个 木棍依次放入四桶
        for (int i=0;i<4;i++) {
            if (matchs[ind] > arr[i]) { //放不下
                continue;
            }
            // 能放下
            if (arr[i] == matchs[ind] || arr[i] >= matchs[ind] + matchs[0]) {
                // 当前桶 >=  当前棍子+ 最短棍子的长度
                arr[i] -= matchs[ind];
                // bool ok = 
                if (dfs(ind -1, arr,matchs)) {
                    return true;
                }
                arr[i] += matchs[ind];
            }
        }
        return false;
    }

};
