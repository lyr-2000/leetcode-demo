3. 无重复字符的最长子串

Description

Difficulty: 中等

Related Topics: 哈希表 , 字符串 , 滑动窗口

给定一个字符串 s ，请你找出其中不含有重复字符的 最长子串的长度。

示例 1:

1
2
3

输入: s = "abcabcbb"
输出: 3 
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。

示例 2:

1
2
3

输入: s = "bbbbb"
输出: 1
解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。

示例 3:

1
2
3
4

输入: s = "pwwkew"
输出: 3
解释: 因为无重复字符的最长子串是 "wke"，所以其长度为 3。
     请注意，你的答案必须是 子串 的长度，"pwke" 是一个子序列，不是子串。

提示：

  • 0 <= s.length <= 5 * 104
  • s 由英文字母、数字、符号和空格组成

Solution

Language: C++

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if(s=="") return 0;
        int l=1,r = s.size();
        while(l<r) {
            int m = (l+r+1)/2;
            if(check(m,s)) l=m;
            else r=m-1;
        }
        return l;
    }
    bool check(int m,string &s) {
        int cnt[256] = {0},k=0;
        int n=s.size();
        int l=0,r=0;
        for(int i=0;i<n;i++) {
            if(cnt[s[i]] == 0) k++;
            cnt[s[i]] ++;
            if(i>=m) {
                cnt[s[i-m]]--;
                if(cnt[s[i-m]] == 0) k--;
            }
            if(k==m) return true;
        }
        return false;
    }
};

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if(s=="") return 0;
        int n = s.size();
        int mp[256]={0};
        int k=0;
        int j=0;
        for(int i=0;i<n;i++) {
            if(mp[s[i]]) {
                while(mp[s[i]]) {
                    mp[s[j++]]--;
                }
            }
            mp[s[i]]++;
            k = max(k,i-j+1);
        }
        return k;
    }
   
};
